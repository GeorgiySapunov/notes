#+begin_src python
import os
import sys
import argparse
from dotenv import load_dotenv
from openai import OpenAI

def main() -> int:
    # 1) Загружаем переменные окружения (.env не обязателен)
    load_dotenv()

    # 2) Считываем аргументы запуска
    parser = argparse.ArgumentParser(description="Первый запрос к LLM")
    parser.add_argument(
        "-q",
        "--query",
        default="Привет, бот! Назови столицу Франции.",
        help="Текст запроса к модели (по умолчанию — простой вопрос)",
    )
    args = parser.parse_args()

    # 3) Создаём клиента OpenAI (v1.x)
    client = OpenAI(
        api_key=os.getenv("OPENAI_API_KEY"),
        # Если используете прокси, задайте свой базовый URL
        base_url=os.getenv("OPENAI_API_BASE", "https://api.openai.com/v1"),
        # при желании можно задать request_timeout=15 для всех вызовов
    )

    # 4) Делаем минимальный запрос (Responses API)
    resp = client.responses.create(
        model=os.getenv(
            "OPENAI_MODEL", "gpt-4o"
        ),  # подставьте свою модель при необходимости
        instructions="Ты дружелюбный ассистент. Отвечай кратко и по делу.",
        input=args.query,
        # temperature=0.0,
        # max_output_tokens=200,
        # timeout=15,  # чтобы не зависать при долгом ответе
    )

    # 5) Печатаем ответ (готовый текст)
    print(resp.output_text.strip())

    # 6) (необязательно) печатаем число токенов, если оно доступно
    usage = getattr(resp, "usage", None)
    total = getattr(usage, "total_tokens", None) if usage is not None else None
    if isinstance(total, int):
        print(f"\n(токены: {total})")

    return 0


if __name__ == "__main__":
    sys.exit(main())

- =ChatOpenAI= – класс LangChain для чат-моделей OpenAI. Мы указали =model_name= и
  temperature. Он автоматически возьмет =OPENAI_API_KEY= из переменной окружения
  (если она установлена), а также =OPENAI_API_BASE= – то есть интеграция с прокси
  тоже сохраняется (LangChain внутри использует openai пакет, поэтому наши
  настройки через =os.getenv= действуют). Мы могли бы явно передать
  =openai_api_key = "..."= и =openai_api_base = "..."= в конструктор, но если они
  настроены в окружении, то это не нужно.
- Метод =invoke()= у =ChatOpenAI= принимает строку – ваш пользовательский запрос – и
  возвращает строку-ответ от модели. За кулисами LangChain сам сформировал
  =messages= с одной ролью =user=, вызвал =openai.ChatCompletion.create= и вернул
  =content= ответа. То есть =chat.invoke("Какой сегодня день недели?")= примерно
  эквивалентно тому, что мы делали руками ранее, только короче.
- Мы сразу печатаем response. Модель, скорее всего, ответит чем-то вроде:
  "Сегодня среда." (или другой день, модель не знает реального времени, но
  попробует угадать или скажет, что не имеет доступа к календарю – зависит от её
  знаний на момент обучения).

from langchain.agents.middleware import before_model
from langchain.agents import AgentState

@before_model
def trim_messages(state: AgentState, runtime) -> dict:
    """Оставить только последние 10 сообщений"""
    messages = state["messages"]
    if len(messages) <= 10:
        return None  # Изменений нет

    # Сохраняем первое (системное) и последние 9
    return {
        "messages": [messages[0]] + messages[-9:]
    }

    tools=[],
    checkpointer=InMemorySaver(),
    middleware=[trim_messages],  # Добавляем middleware

#+begin_src python
import os
from dotenv import load_dotenv
load_dotenv()

import logging
from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder
from langchain_core.chat_history import InMemoryChatMessageHistory # self.store[session_id] = InMemoryChatMessageHistory()
from langchain_core.runnables.history import RunnableWithMessageHistory # self.chain_with_history = RunnableWithMessageHistory(
from langchain_openai import ChatOpenAI

# Создаём класс для CLI-бота
class cli_bot():
    def __init__(self, model_name, system_prompt="Ты полезный ассистент."):
        # Создаём модель
        self.chat_model = ChatOpenAI(
            model_name=model_name,
            temperature=0,
            request_timeout=15
        )

        # Создаём Хранилище истории
        self.store = {}

        # Создаем шаблон промпта
        self.prompt = ChatPromptTemplate.from_messages([
            ("system", system_prompt), # Добавим возможность менять системный промпт
            MessagesPlaceholder(variable_name="history"), # Создаем placeholder для истории чата. variable_name: Название переменной, хранящей историю сообщений.
            ("human", "{question}"),
        ])

        # Создаём цепочку
        self.chain = self.prompt | self.chat_model

        # Создаём цепочку с историей
        self.chain_with_history = RunnableWithMessageHistory(
            self.chain, # Цепочка с историей
            self.get_session_history, # метод для получения истории
            input_messages_key="question", # ключ для вопроса
            history_messages_key="history", # ключ для истории
        )

        # Настраиваем логирование в файл
        logging.basicConfig(
            filename="chat_session.log", level=logging.INFO,
            format="%(asctime)s [%(levelname)s] %(message)s", datefmt="%Y-%m-%d %H:%M:%S",
            encoding="utf-8"
        )


    # Метод для получения истории по session_id
    def get_session_history(self, session_id: str):
        if session_id not in self.store:
            self.store[session_id] = InMemoryChatMessageHistory()
        return self.store[session_id]

    def __call__(self, session_id):
        print("Чат-бот запущен! Можете задавать вопросы. \n - Для выхода введите 'выход'.\n - Для очистки контекста введите 'сброс'.\n")
        logging.info("=== New session ===")
        while True:
            try:
                user_text = input("Вы: ").strip()
            except (KeyboardInterrupt, EOFError):
                print("\nБот: Завершение работы.")
                break
            if not user_text:
                continue

            logging.info(f"User: {user_text}")
            msg = user_text.lower()
            if msg in ("выход", "стоп", "конец"):
                print("Бот: До свидания!")
                logging.info("Пользователь завершил сессию. Сессия окончена.")
                break
            if msg == "сброс":
                if session_id in self.store:
                    del self.store[session_id]
                print("Бот: Контекст диалога очищен.")
                logging.info("Пользователь сбросил контекст.")
                continue

            try:
                responce = self.chain_with_history.invoke(
                    {"question": user_text},
                    {"configurable": {"session_id": session_id}}

                logging.error(f"[error] {e}")
                print(f"[Ошибка] {e}")
                continue

            # Форматируем и выводим ответ
            bot_reply = responce.content.strip()
            logging.info(f"Bot: {bot_reply}")

** Few-shot метод
*Few-shot* prompting основан на том, что в запрос (prompt) к модели включаются
несколько примеров входных данных и ожидаемых ответов.

Например, если нужно, чтобы модель выдавала ответ в виде таблицы, списка или
текста в определённом стиле, достаточно привести 2–3 готовых примера — и новый
ответ, с высокой вероятностью, будет следовать этому шаблону.

*Few-shot* подсказки особенно полезны, когда обычного *zero-shot* режима (только
инструкция без примеров) недостаточно

def chat_with_reminder(response, remind_every=6):
    response = response if isinstance(response, str) else response.content
    history = get_session_history('support')
    non_system = [m for m in history.messages if not isinstance(m, SystemMessage)]

    if len(non_system) > 0 and len(non_system) % remind_every == 0:
        history.add_message(SystemMessage(content=f"НАПОМИНАНИЕ: {support}"))
        print(f"⚠️ Добавлено напоминание роли (сообщение #{len(non_system)})")

    response = chain_with_history.invoke(
        {"system": support, "user_input": response},
        {"configurable": {"session_id": 'support'}})
    return response

import os
from dotenv import load_dotenv
from langchain_openai import ChatOpenAI
from langchain_core.chat_history import BaseChatMessageHistory
from langchain_core.messages import BaseMessage, SystemMessage, HumanMessage
import tiktoken

load_dotenv()

llm = ChatOpenAI(model_name=MODEL, temperature=0.9)

# Кастомная память с автоматическим добавлением системного промпта
class MemoryWithSystemPrepend(BaseChatMessageHistory):
    def __init__(self, system_prompt: str, max_tokens: int = 4000):
        self.system_prompt = system_prompt
        self._messages = []  # Храним только диалог (без system)
        self.max_tokens = max_tokens
        self.encoder = tiktoken.get_encoding("cl100k_base")

    def count_tokens(self, messages):
            total = 0
            for msg in messages:
                total += len(self.encoder.encode(msg.content))
            return total

    @property
    def messages(self):
        """Подготовка истории: prepend system + усечение до max_tokens"""
        history = [SystemMessage(content=self.system_prompt)] + self._messages

        while self.count_tokens(history) > self.max_tokens and len(history) > 1:
            history.pop(1)

        return history

    def add_message(self, message: BaseMessage):
        if not isinstance(message, SystemMessage):
            self._messages.append(message) # добавляем сообщения только в диалог

    def clear(self):
        self._messages = []


# Использование
support_memory = MemoryWithSystemPrepend("Ты опытный ассистент поддержки интернет-провайдера. Ты всегда вежлив и доброжелателен. Отвечай не длиннее 100 символов.")
client_memory = MemoryWithSystemPrepend("Ты раздражённый клиент у которого не работает интернет. Стиль общения грубый. Отвечай не длиннее 100 символов.")

# Простой чат без RunnableWithMessageHistory для наглядности
def simple_chat(user_input: str, memory: MemoryWithSystemPrepend):
    memory.add_message(HumanMessage(content=user_input))
    response = llm.invoke(memory.messages)
    memory.add_message(response)

    return response.content

** Метрики качества RAG
*** Метрики поиска (Retrieval):
- Context Precision (точность контекста): Какая доля найденных документов действительно относится к вопросу.
- Context Recall (полнота контекста): Насколько полно найденные документы покрывают факты, необходимые для ответа.
- Precision@K: Сколько из топ-k найденных документов являются релевантными.
- Recall@k: Cколько от всех релевантных документов нашлось в топ-k.
- F1@k: комбинированная метрика поиска, гармоническое среднее precision и recall. Полезна, когда важно и то, и другое сбалансировано.

**** Код для оценки Precision@K и Recall@K
Precision@K: Сколько из топ-k найденных документов являются релевантными.
Низкий precision – значит, много лишнего. Возможно, chunk слишком большой...
- Recall@k: Cколько от всех релевантных документов нашлось в топ-k.
Низкий recall – значит, retriever недостает нужные доки. Решения: повысить k, улучшить эмбеддинг...

В поиске эти метрики противостоят. Если настроить retriever возвращать больше документов (увеличить k ) то:
- recall растет (не упустим нужные документы)
- precision падает (среди большего списка документов, доля релевантных меньше )

# Загружаем retriever
embed_model = HuggingFaceEmbeddings(model_name="all-MiniLM-L6-v2")

retriever = vector_store.as_retriever(search_kwargs={"k": 2})

# Оценка качества поиска
k = 2

precision_total = 0
recall_total = 0
for question, true_docs in ground_truth_docs.items():
    found_docs = retriever.invoke(question)
    found_ids = [d.metadata["source"] for d in found_docs[:k]]

    # Считаем, сколько релевантных нашли
    relevant_found = sum(1 for doc_id in true_docs if doc_id in found_ids)
    precision = relevant_found / k
    precision_total += precision
    recall = relevant_found / len(true_docs) if true_docs else 0
    recall_total += recall

    print(f"Вопрос: {question}")
    print(f"  Эталонные документы: {true_docs}")
    print(f"  Найденные документы: {found_ids}")
    print(f"  Precision@{k}: {precision:.2f}")
    print(f"  Recall@{k}: {recall:.2f} ({relevant_found}/{len(true_docs)})\n")

precision_avg = precision_total / len(ground_truth_docs)
print(f"Средняя Precision@{k}: {precision_avg:.2f}")
recall_avg = recall_total / len(ground_truth_docs)
print(f"Средняя Recall@{k}: {recall_avg:.2f}")

*** Метрики ответа (Generation):
- Faithfulness — измеряет, насколько ответ опирается исключительно на предоставленный контекст и не содержит галлюцинаций.
- AnswerRelevancy — оценивает соответствие ответа исходному вопросу и его информативность.
- ContextRecall — определяет, насколько извлечённый retriever контекст покрывает все необходимые факты, содержащиеся в эталонном ответе.
- ContextPrecision — измеряет долю действительно полезной информации среди извлечённых документов (насколько retriever не приносит “лишние” куски текста).
- AnswerCorrectness — сравнивает фактическую корректность ответа с эталонным ответом.
- ContextUtilization — определяет, насколько ответ реально использует информацию из контекста, а не опирается на внешние знания.
- AnswerSimilarity — измеряет степень совпадения ответа с эталонным, не требуя буквального совпадения.

def evaluate_qa_with_llm(query: str, prediction: str, reference: str) -> dict:
    prompt = f"""Ты эксперт по оценке качества ответов.

Вопрос: {query}
Ожидаемый ответ: {reference}
Полученный ответ: {prediction}

Оцени, насколько полученный ответ соответствует ожидаемому по смыслу.
Ответь ТОЛЬКО одним словом: CORRECT или INCORRECT"""

    response = llm.invoke(prompt)
    verdict = response.content.strip().upper()
    score = 1.0 if verdict == "CORRECT" else 0.0

    return score, verdict

scores_llm = []
for example, prediction in zip(examples, predictions):
    score, verdict = evaluate_qa_with_llm(
        query=example["query"],
        prediction=prediction["result"],
        reference=example["answer"]
    )

*** Гибридный поиск
Объединение лексического (BM25) и семантического (эмбеддинги) поиска

def bm25_candidates(query, k):
    toks = tokenize(query) # токенезируем вопрос
    scores = bm25.get_scores(toks)  # получаем скор
    # Сортируем индексы документов по убыванию скоринга и выбираем top-k
    top_idx = np.argsort(scores)[::-1][:k]

    res = {}
    for i in top_idx:
        doc_id = list(id2doc.keys())[i]
        res[doc_id] = float(scores[i])
    return res

**** Получение скора из векторного хранилища
- В langchain метод =similarity_search_with_score()= у векторного хранилища
  возвращает Tuple с документом и расстоянием типа float и чем оно меньше, тем
  релевантней этот документ, поэтому нам нужно это значение инвертировать для
  нормализации этого скора к диапозону [0,1], где 1 это наиболее релевантный, а
  0 соответственно наименее
- На выходе мы, как и в случае BM25-поиска, получаем словарь той же структуры, и
  теперь нам удобно их совмещать.
#+begin_src python
from typing import Dict
import numpy as np

def vector_candidates(query, k):
    # Выполняем векторный поиск: получаем пары (документ, расстояние)
    # similarity_search_with_score возвращает dist — чем меньше, тем ближе документ
    docs_scores = vector_store.similarity_search_with_score(query, k=k)

    # Конвертируем расстояние в псевдо-скор: 1 / (1 + dist)
    # Чем меньше dist → тем больше score. Значения будут в (0..1]
    tmp = {d.metadata.get("id"): 1.0 / (1.0 + float(dist)) for d, dist in docs_scores}

    # Нормализуем все скоры в диапазон [0..1], чтобы можно было
    # далее безопасно комбинировать их с BM25
    return normalize(tmp)


def normalize(scores, reverse=False):
    if not scores:
        return {}
    vals = np.array(list(scores.values()), dtype=float)
    if reverse:
        vals = -vals
    lo, hi = float(vals.min()), float(vals.max())
    if hi == lo:
        return {k: 1.0 for k in scores}
    keys = [key for key,_ in scores.items()]
    norm = (vals - lo) / (hi - lo)
    # Собираем обратно в словарь: doc_id → нормализованный скор
    return {key: float(value) for key, value in zip(keys, norm)}

**** Reciprocal Rank Fusion (RRF)
Алгоритм, объединяющий рейтинги документов из двух списков, суммируя вклад от
рангов. Например, каждому документу присваивается суммарный скор по формуле
=RRF_score = Σ (1 / (k + rank))= для каждого ранжированного списка. RRF избегает
сложной нормализации баллов и эффективно комбинирует результаты от BM25 и
векторного поиска.
#+begin_src python
from typing import Dict
from collections import defaultdict

def to_ranks(scores, higher_better=True):
        items = sorted(scores.items(), key=lambda kv: kv[1], reverse=higher_better)
        return {doc_id: rank+1 for rank, (doc_id, _) in enumerate(items)}

def rrf(bm25, vec, rrf_k=60):
    # превращаем оценки в ранги (1 — лучший)
    ranks_bm25 = to_ranks(bm25, higher_better=True)
    ranks_vec  = to_ranks(vec,  higher_better=True)
    # объединяем носители
    all_ids = set(ranks_bm25) | set(ranks_vec)
    fused = defaultdict(float)
    for doc_id in all_ids:
        if doc_id in ranks_bm25:
            fused[doc_id] += 1.0 / (rrf_k + ranks_bm25[doc_id])
        if doc_id in ranks_vec:
            fused[doc_id] += 1.0 / (rrf_k + ranks_vec[doc_id])
    return dict(fused)
#+end_src
**** Взвешенное объединение баллов
​​Оценки схожести предварительно нормализуются, потому как масштабы и знаки разных
ретриверов разные и если этого не сделать, то сравнивать их в лоб будет не
корректно, один просто перебьёт другой. Затем вычисляется итоговый скор по
формуле: =α * score_BM25 + (1 - α) * score_vector=. Если α=0.5, вклад равный;
более высокое α усиливает роль точного совпадения ключевых слов.
#+begin_src python
def weighted(bm25, vec, alpha):
    bm25_n = normalize(bm25) # выше — лучше
    # vec уже нормализован в [0..1]
    vec_n = normalize(vec)
    all_ids = set(bm25_n) | set(vec_n)
    fused = {}
    a = float(alpha)
    for doc_id in all_ids:
        fused[doc_id] = a * bm25_n.get(doc_id, 0.0) + (1.0 - a) * vec_n.get(doc_id, 0.0)
    return fused

*** Cross-encoder reranking
Cross-encoder - это модель (обычно на базе трансформера), которая принимает на
вход пару «запрос–документ» и выдаёт степень их релевантности. В отличие от
*bi-encoder*, где запрос и документ превращаются в эмбеддинги раздельно,
кросс-энкодер совместно обрабатывает оба текста, что позволяет учесть более
тонкие взаимосвязи. Благодаря этому *cross-encoder* способен более точно оценить,
отвечает ли конкретный фрагмент документа на вопрос пользователя.

В pipeline гибридного поиска cross-encoder используется для повторного ранжирования (reranking):
- Сначала, гибридный поиск (BM25 + embeddings) извлекает расширенный список
  кандидатов – например по 50 документов от каждого метода, итого до ~100
  уникальных чанков.
- Затем cross-encoder прогоняется для каждой пары «запрос–документ» из этого
  списка и присваивает каждому документу новую оценку релевантности (как
  правило, вероятность того, что документ содержит ответ).
- Кандидаты сортируются по этому новому скору и отбираются топ N (например,
  топ-5 или 10), для подачи в LLM.

Для реранка можно использовать готовые модели, натренированные на задаче
поискового ранжирования. Существуют публичные cross-encoder модели (например,
из библиотеки =sentence-transformers=), возвращающие скор релевантности.
#+begin_src python
from sentence_transformers import CrossEncoder

class SimpleReranker:
    def __init__(self, model_name):
        """
        Args:
            model_name: Название модели cross-encoder
        """
        self.model = CrossEncoder(model_name)

    def rerank(self, query, documents, top_n=3):
        """
        Args:
            query: Поисковый запрос
            documents: Список документов для переоценки
            top_n: Сколько документов возвращает
        Returns:
            Список топ-N документов после rerank
        """

        if not documents:
            return []

        # Подготавливаем пары (query, document)
        pairs = [[query, doc.page_content] for doc in documents]

        # Получаем скоры релевантности
        scores = self.model.predict(pairs)

        # Сортируем документы по скору
        doc_score_pairs = list(zip(documents, scores))
        doc_score_pairs.sort(key=lambda x: x[1], reverse=True)

        # Отбираем топ-N документов (score записываем в метаданные)
        result = []
        for doc, score in doc_score_pairs[:top_n]:
            doc.metadata = doc.metadata or {}
            doc.metadata["rerank_score"] = float(score)
            result.append(doc)

        return result

#+begin_src python
from langchain_core.retrievers import BaseRetriever
import hashlib

class HybridRerankerRetriever(BaseRetriever):
    first_retriever: BaseRetriever
    second_retriever: BaseRetriever
    reranker: SimpleReranker
    k: int = 3

    def _dedup_docs(self, docs):
        '''объединение результатов с дедупликацией'''
        seen = set()
        result = []
        for d in docs:
            content = d.page_content
            uid = hashlib.md5(content.encode("utf-8")).hexdigest()
            if uid not in seen:
                seen.add(uid)
                result.append(d)
        return result

    def _get_relevant_documents(self, query):
        """Вызывает два других ретривера
        Объединяет их результаты, удаляя дубликаты методом _dedup_docs()
        Вызывает реранкер, который возвращает топ k выдачи"""
        first_docs = self.first_retriever.invoke(query)
        second_docs = self.second_retriever.invoke(query)
        merged = self._dedup_docs(first_docs + second_docs)

        return self.reranker.rerank(query, merged, self.k)

class ScoredRetriever(BaseRetriever):
    vector_store: Any
    search_kwargs: Dict

    def _get_relevant_documents(self, query):
        docs_scores = self.vector_store.similarity_search_with_score(
            query,
            **self.search_kwargs

        docs_with_scores = []
        for doc, dist in docs_scores:
            doc.metadata["score"] = 1.0 / (1.0 + float(dist))
            docs_with_scores.append(doc)

        return docs_with_scores

def check_confidence(inputs, threshold=0.55):

            'docs': inputs['docs'],
            'used_llm': False

    return inputs

from langchain.tools import tool
import requests

def geocode(city: str):
    '''Получает координаты по названию города'''
    resp = requests.get(
        "https://nominatim.openstreetmap.org/search",
        params={"q": city, "format": "json", "limit": 1},
        headers={"User-Agent": "weather-agent/1.0"}
    )
    data = resp.json()
    if not data:
        return None
    return float(data[0]["lat"]), float(data[0]["lon"])

@tool("get_weather")
def get_weather(city: str) -> str:
    """Возвращает текущую температуру в указанном городе."""
    coords = geocode(city)
    if not coords:
        return f"Город «{city}» не найден."
    lat, lon = coords
    resp = requests.get(
        "https://api.open-meteo.com/v1/forecast",
        params={"latitude": lat, "longitude": lon, "current_weather": True},
        timeout=5
    )
    data = resp.json()
    cw = data.get("current_weather")
    if not cw:
        return "Не удалось получить текущую погоду."
    return f"В городе {city} сейчас {cw['temperature']}°C."

@tool("convert_currency")
def convert_currency(amount: float, from_currency: str, to_currency: str) -> str:
    """
    Конвертирует сумму из from_currency в to_currency по актуальному курсу.

    Пример: amount=10, from_currency='USD', to_currency='RUB'
    """
    from_currency = from_currency.upper()
    to_currency = to_currency.upper()

    try:
        resp = requests.get("https://api.fxratesapi.com/latest", timeout=5)
        resp.raise_for_status()
        data = resp.json()
        rates = data.get("rates")
        if not rates:
            return f"Не удалось получить курсы валют: {data}"

        if from_currency not in rates:
            return f"Неизвестная валюта-источник: {from_currency}"
        if to_currency not in rates:
            return f"Неизвестная валюта-назначение: {to_currency}"

        # конвертация через базовую валюту API (USD)
        amount_in_usd = amount / rates[from_currency]  # переводим в USD
        converted_amount = round(amount_in_usd * rates[to_currency], 2)

        return f"{amount} {from_currency} ≈ {converted_amount} {to_currency} (курс {to_currency}: {rates[to_currency]}, курс {from_currency}: {rates[from_currency]})"

    except Exception as e:
        return f"Ошибка при запросе курса: {e}"

=ToolRetryMiddleware= автоматически повторяет вызовы инструментов при ошибках
Итоговая задержка рассчитывается по формуле =delay = initial_delay * (backoff_factor ** attempt) * jitter=
#+begin_src python
from langchain.agents import create_agent
from langchain.agents.middleware import ToolRetryMiddleware

agent = create_agent(
    model=model,
    tools=[get_weather, convert_currency],
    middleware=[
        ToolRetryMiddleware(
            max_retries=3, # если ошибка, то пробуем ещё 3 раза
            backoff_factor=2.0, # экспоненциальный backoff. Это означает, что
            # время ожидания растёт в столько-то раз в геометрической прогрессии.
            # При значении 2 с каждым повтором будет удваиваться: через 1 секунду,
            # потом через 2, 4, 8 и т.д. Это даёт сервису время восстановиться, и
            # снижает нагрузку
            initial_delay=1.0, # задержка перед изначальной повторной попыткой в секундах
            jitter=True, # ±25% случайного разброса в задержках. Если у нас 100
            # параллельных запросов, и все провалились, то через 1 секунду они все
            # разом повторятся и мы опять перегрузим удалённый сервер запросами.
            # Джиттер решает это, то есть, 2-я попытка не ровно через 1 сек, а
            # случайная величина от 0.75 до 1.25 и так далее с каждым повтором.
            # Так повторные запросы размазываются по времени, уменьшая шанс нового
            # массового сбоя. Это стандартная рекомендация от AWS и других крупных
            # компаний. По умолчанию этот параметр включен True
            retry_on=(ConnectionError, TimeoutError),
            on_failure="return_message", # список ошибок, на которые реагируем, ConnectionError,TimeoutError и т.д.
            max_delay=60, # Если сервис постоянно падает (например, API
            # недоступно), нет смысла дергать его сотню раз – это лишь тратит
            # время. Реализуется простая логика: после 3 подряд неудачных вызовов
            # блокируем дальнейшие попытки на минуту (Значение по умолчанию). Это
            # предотвращает: накопление висящих потоков, лишнюю нагрузку на уже и
            # так проблемный сервис, cascades (когда сбой одного компонента тянет
            # за собой сбой всей системы).
        ),
    ],
)

class FallbackModelMiddleware(AgentMiddleware):
    def __init__(self, backup_model, fail_message="сервис недоступен"):
        self.backup_model = backup_model
        self.fail_message = fail_message

    def wrap_model_call(self, request: ModelRequest, handler):
        # Основная модель
        try:
            return handler(request)
        except Exception:
            # тут можно реализовывать логику на случай отказа основной модели
            # например popup сообщение о смене модели в UI чата и т.п.
            pass

        try:
            new_request = request.override(model=self.backup_model)
            return handler(new_request)
        except Exception:
            # Если не работают обе, отдаём сообщение об этом
            return ModelResponse(
                result=[AIMessage(content=self.fail_message)]

import re

def extract_date_from_url(url):
    """Пытается извлечь дату из URL"""
    # Паттерны типа /2025/11/20/ или /2025-11-20/
    patterns = [
        r'/(\d{4})/(\d{2})/(\d{2})/',
        r'/(\d{4})-(\d{2})-(\d{2})/',
        r'date=(\d{4})-(\d{2})-(\d{2})',
    ]

    for pattern in patterns:
        match = re.search(pattern, url)
        if match:
            year, month, day = match.groups()
            return f"{day}.{month}.{year}"

    return None

def extract_date_from_title(title):
    """Пытается извлечь дату из заголовка"""
    # Паттерны типа "20 Nov 2025" или "2025-11-20"
    patterns = [
        r'(\d{1,2})\s+(Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)\s+(\d{4})',
        r'(\d{4})-(\d{2})-(\d{2})',
    ]

    for pattern in patterns:
        match = re.search(pattern, title)
        if match:
            return match.group(0)

    return None

import requests
from bs4 import BeautifulSoup

def fetch_publication_date(url, timeout=5):
    """Извлекает дату публикации из HTML страницы"""
    try:
        response = requests.get(url, timeout=timeout)
        soup = BeautifulSoup(response.content, 'html.parser')

        # Ищем в meta тегах
        meta_dates = [
            soup.find('meta', property='article:published_time'),
            soup.find('meta', {'name': 'publication_date'}),
            soup.find('meta', {'name': 'date'}),
            soup.find('time')
        ]

        for meta in meta_dates:
            if meta and meta.get('content'):
                return meta.get('content')
            elif meta and meta.get('datetime'):
                return meta.get('datetime')

        return None
    except:
        return None

import time

def estimate_tokens(text, model_name=None):
    """Оценка токенов, точная если известна модель или грубая (1 токен ≈ 4 символа)"""
    try:
        import tiktoken
        if model_name:
            enc = tiktoken.encoding_for_model(model_name)
        else:
            enc = tiktoken.get_encoding("cl100k_base")
        return len(enc.encode(text))
    except Exception:
        return len(text) // 4

@wrap_model_call
def my_first_wrapper(request, handler):
    """Обёртка — логируем токены и время ответа модели"""

    # Собираем весь текст из сообщений и замеряем токены
    input_text = "".join(getattr(msg, "content", "") for msg in request.messages)
    input_tokens_est = estimate_tokens(input_text, MODEL)

    # Вызываем модель замерив время
    start = time.time()
    response = handler(request)
    elapsed = time.time() - start

    result = response.result[0] if response.result else None
    if result:
        metadata = getattr(result, "usage_metadata", None)

        # если есть метаданные, читаем из них
        if metadata and isinstance(metadata, dict):
            input_tokens = metadata.get("input_tokens", input_tokens_est)
            output_tokens = metadata.get("output_tokens", 0)
            print(f"Входных токенов: {input_tokens}, Выходных токенов: {output_tokens}")
        else:
            # если нету, то замеряем
            output_text = str(getattr(result, "content", ""))
            output_tokens = estimate_tokens(output_text, MODEL)
            print(f"Входных токенов: {input_tokens_est}, Выходных токенов: {output_tokens}")

        print(f"Модель ответила за {elapsed:.2f} секунд")

    return response

*** Пример кода: кэширование ответов LLM
У данного класса есть методы =before_model()= и =after_model()= и они соответственно
выполняют некие действия до и после вызова модели. Также у метода можно
реализовать декоратор =@hook_config(can_jump_to = ["end"])=, который может при
определённом условии перекинуть нас сразу в конец, то есть миновав вызов модели.

Другие варианты перехода: к инструментам =tools=, к модели =model=, в конец =end=.

В =before_model()= мы перехватываем то, что идёт в модель, парсим текст сообщения
и если его хэш есть в нашем словаре, то мы просто перескакиваем в конец без
вызова модели и отдаём значение этого словаря, обернув его в объект =AIMessage()=
класса.

Если такого хэша нет, то просто идём дальше и генерируем ответ модели, который в
=after_model()= мы сохраняем в кэш с ключом, которым является хэш нашего
вопроса.

class HookCacheMiddleware(AgentMiddleware):
    def __init__(self, path: str = "llm_cache.json"):
        self.path = path
        self._load_cache()

    def _load_cache(self):
        if os.path.exists(self.path):
            try:
                with open(self.path, 'r', encoding='utf-8') as f:
                    self.cache = json.load(f)
            except Exception:
                self.cache = {}
        else:
            self.cache = {}

    def _save_cache(self):
        with open(self.path, "w", encoding="utf-8") as f:
            json.dump(self.cache, f, ensure_ascii=False, indent=2)

    def _hash_key(self, text: list[Any]) -> str:
        return hashlib.md5(text.lower().strip().encode("utf-8")).hexdigest()

    @hook_config(can_jump_to=["end"])
    def before_model(self, state, runtime):
        input_text = [m for m in state.get("messages") if isinstance(m, HumanMessage)][-1].content
        self.key = self._hash_key(input_text)

        if self.key in self.cache:
            cached_answer = self.cache[self.key]
            return {"jump_to": "end", "messages": [AIMessage(content=cached_answer)]}
        return None

    def after_model(self, state, runtime):
        answer_text = [msg for msg in state.get("messages") if isinstance(msg, AIMessage)][-1].content
        self.cache[self.key] = answer_text
        self._save_cache()
        return None


model = ChatOpenAI(model_name=MODEL, temperature=0)
agent = create_agent(
    model=model,
    tools=[],
    middleware=[HookCacheMiddleware()]

from langchain_community.vectorstores import FAISS
import hashlib
from typing import Dict
from langchain_core.retrievers import BaseRetriever
import time

class CachedRetriever(BaseRetriever):
    vectorstore: FAISS
    cache: Dict = {}
    ttl_seconds: int = 3600
    k: int = 3

    def _hash_query(self, query: str) -> str:
        return hashlib.md5(query.lower().strip().encode()).hexdigest()

    def _get_relevant_documents(self, query):
        key = self._hash_query(query)
        now = time.time()

        if key in self.cache:
            timestamp, results = self.cache[key]
            age = int(now - timestamp)

            if age < self.ttl_seconds:
                return results
            else:
                del self.cache[key]

        results = self.vectorstore.similarity_search(query, k=self.k)
        self.cache[key] = (now, results)
        return results

def llm_with_context(question: str) -> str:
    context = ""

    # tracker
    match = re.search(r"Ord-No:\d{4}", question)
    if match:
        context += track(match.group(0)) + "\n"

    # retriever
    retriever_keywords = ["о компании", "услуги", "продукт", "интеграция", "документация", "api", "sdk", "описание", "функции"]
    if any(term in question.lower() for term in retriever_keywords):
        context += retriever(question) + "\n"

    # web_search
    web_search_keywords = ["курс", "погода", "новости", "сегодня", "цены", "инфляция", "события"]
    if any(term in question.lower() for term in web_search_keywords):
        context += web_search(question) + "\n"

    # Создаём prompt template с контекстом
    template = ChatPromptTemplate.from_messages([
        ("system", "Ты полезный ассистент, отвечай пользователю опираясь на контекст, если он есть."),
        ("human", "Вопрос пользователя: {question}\nКонтекст: {context}")
    ])

    chain = template | llm
    response = chain.invoke({"question": question, "context": context})

    return response.content

@tool
def multiply(a: int, b: int) -> int:
    """Умножение 'a' на 'b'"""
    return a * b

@tool
def add(a: int, b: int) -> int:
    """Сложение `a` и `b`"""
    return a + b

@tool
def divide(a: int, b: int) -> float:
    """Деление `a` на `b`"""
    return a / b

from langgraph.graph import MessagesState

class CustomMessagesState(MessagesState):
    llm_calls: int

# Создаём функцию для использования в узле с вызовом LLM:
from langchain.messages import SystemMessage

def llm_call(state: CustomMessagesState):
    system_prompt = "Ты полезный ассистент, помогающий с арифметическими операциями"
    return {
        "messages": [model_with_tools.invoke([SystemMessage(content=system_prompt)] + state["messages"])],
        "llm_calls": state.get('llm_calls', 0) + 1
    }

# Создаём функцию для узла вызова инструментов:
from langchain.messages import ToolMessage

def tool_node(state: CustomMessagesState):
    result = []
    for tool_call in state["messages"][-1].tool_calls:
        tool = tools_by_name[tool_call["name"]]
        observation = tool.invoke(tool_call["args"])
        result.append(ToolMessage(content=observation, tool_call_id=tool_call["id"]))
    return {"messages": result}

from langgraph.graph import StateGraph, START, END

def should_continue(state: CustomMessagesState):
    last_message = state["messages"][-1]
    # Если LLM вызывает инструмент, роутер переключается на инструмент
    if last_message.tool_calls:
        return "tool_node"

    # Если вызова инструмента нет, то переключаем на выход и отдаём ответ пользователю
    return END

graph.add_node("tool_node", tool_node)

graph.add_conditional_edges(

    should_continue,

graph.add_edge("tool_node", "llm_call")

class CodeState(MessagesState):

# имитациями. Предположим там просто print() и модель возвращает его с ошибкой:
def generate_code(state: CodeState):
    """Генерирует код с ошибкой"""
    return {"code": "print('code with error')"}

def execute_code(state: CodeState):
    """Убирает print() и выводит содержимое"""
    code = state["code"]
    # Убираем print() и кавычки
    if "print(" in code:
        result = code.replace("print(", "").replace(")", "").strip("'\"")
    else:
        result = code
    return {"result": result}

import ast
from pathlib import Path

class CodeValidator(ast.NodeVisitor):
    def __init__(self):
        self.error = None

    def visit_Import(self, node):
        for alias in node.names:
            root = alias.name.split(".")[0]
            if root not in ALLOWED_MODULES:
                self.error = f"❌ Модуль '{root}' не разрешён"
        self.generic_visit(node)

    def visit_ImportFrom(self, node):
        if node.module is None:
            self.error = "❌ Относительные импорты запрещены"
            return
        root = node.module.split(".")[0]
        if root not in ALLOWED_MODULES:
            self.error = f"❌ Модуль '{root}' не разрешён"
        self.generic_visit(node)

    def visit_Call(self, node):
        if isinstance(node.func, ast.Name):
            if node.func.id in FORBIDDEN_NAMES:
                self.error = f"❌ Запрещённая функция: {node.func.id}"

        if isinstance(node.func, ast.Attribute):
            if node.func.attr in FORBIDDEN_NAMES:
                self.error = f"❌ Запрещённый атрибут: {node.func.attr}"
        self.generic_visit(node)

    def visit_Attribute(self, node):
        if node.attr.startswith("__"):
            self.error = "❌ Доступ к dunder-атрибутам запрещён"
        self.generic_visit(node)

    def visit_Constant(self, node):
        if isinstance(node.value, str):
            try:
                p = Path(node.value)
                if p.is_absolute() or "/" in node.value or "\\" in node.value:
                    resolved = (SANDBOX_DIR / p).resolve()
                    if not resolved.is_relative_to(SANDBOX_DIR):
                        self.error = f"❌ Доступ вне sandbox запрещён: {node.value}"
            except Exception:
                pass
        self.generic_visit(node)

def validate_code(code: str) -> tuple[bool, str]:
    try:
        tree = ast.parse(code)
    except SyntaxError as e:
        return False, f"❌ SyntaxError: {e}"

    validator = CodeValidator()
    validator.visit(tree)
    if validator.error:
        return False, validator.error
    return True, "✅ Код безопасен для выполнения в sandbox"

        try:
            result = self.executor.execute_code_blocks([("python", code)])
            return result[0], result[1]

import autogen

_executor = None
def get_executor():
    global _executor
    if _executor is None:
        _executor = autogen.UserProxyAgent(
            name="executor",
            human_input_mode="NEVER",
            code_execution_config={"work_dir": str(WORK_DIR), "use_docker": False}
        )
    return _executor

def execute_code(state: CoderState) -> dict:
    executor = get_executor()
    exit_code, output = executor.execute_code_blocks([("python", state['code'])])

    if exit_code == 0:
        return {"result": output}
    else:
        return {"result": f"[ERROR] {output}"}

import ast
from pathlib import Path

class CodeValidator(ast.NodeVisitor):
    def __init__(self):
        self.error = None

    def visit_Import(self, node):
        for alias in node.names:
            root = alias.name.split(".")[0]
            if root not in ALLOWED_MODULES:
                self.error = f"❌ Модуль '{root}' не разрешён"
        self.generic_visit(node)

    def visit_ImportFrom(self, node):
        if node.module is None:
            self.error = "❌ Относительные импорты запрещены"
            return
        root = node.module.split(".")[0]
        if root not in ALLOWED_MODULES:
            self.error = f"❌ Модуль '{root}' не разрешён"
        self.generic_visit(node)

    def visit_Call(self, node):
        if isinstance(node.func, ast.Name):
            if node.func.id in FORBIDDEN_NAMES:
                self.error = f"❌ Запрещённая функция: {node.func.id}"

        if isinstance(node.func, ast.Attribute):
            if node.func.attr in FORBIDDEN_NAMES:
                self.error = f"❌ Запрещённый атрибут: {node.func.attr}"
        self.generic_visit(node)

    def visit_Attribute(self, node):
        if node.attr.startswith("__"):
            self.error = "❌ Доступ к dunder-атрибутам запрещён"
        self.generic_visit(node)

    def visit_Constant(self, node):
        if isinstance(node.value, str):
            try:
                p = Path(node.value)
                if p.is_absolute() or "/" in node.value or "\\" in node.value:
                    resolved = (SANDBOX_DIR / p).resolve()
                    if not resolved.is_relative_to(SANDBOX_DIR):
                        self.error = f"❌ Доступ вне sandbox запрещён: {node.value}"
            except Exception:
                pass
        self.generic_visit(node)

def validate_code(code: str) -> tuple[bool, str]:
    try:
        tree = ast.parse(code)
    except SyntaxError as e:
        return False, f"❌ SyntaxError: {e}"

    validator = CodeValidator()
    validator.visit(tree)
    if validator.error:
        return False, validator.error
    return True, "✅ Код безопасен для выполнения в sandbox"

def execute_code(state: CoderState) -> dict:
    code = state['code']
    is_safe, message = validate_code(code)
    if not is_safe:
        return {"result": f"ОШИБКА БЕЗОПАСНОСТИ: {message}"}
    try:
        executor = get_executor()
        exit_code, output = executor.execute_code_blocks([("python", code)])

        if exit_code == 0:
            return {"result": output}
        else:
            return {"result": f"Ошибка выполнения:\n{output}"}
    except Exception as e:
        error_msg = str(e)
        return { "result": f"Исключение: {error_msg}"}
